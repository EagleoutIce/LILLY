\def\LILLYcommand{
    \@ifstar%
        \LILLYcommand@S %% star
        \LILLYcommand@N %% nostar
}

\def\LILLYcommand@S#1{
    \let#1\outer\renewcommand*#1
}
\def\LILLYcommand@N#1{
    \let#1\outer\renewcommand#1
}
