\renewcommand{\enum}[1]{\begin{enumerate}\narrowitems  #1 \end{enumerate}  }
\renewcommand{\liste}[1]{\begin{enumerate}[label=\roman*)]\narrowitems  #1 \end{enumerate}  }

\renewcommand{\xa}{\overline{x_1}\tab[0.04cm]}
\renewcommand{\xb}{\overline{x_2}\tab[0.04cm]}
\renewcommand{\xc}{\overline{x_3}\tab[0.04cm]}

\renewcommand{\rddots}{\reflectbox{$\ddots$}}

\renewcommand{\crossAT}[1]{\node at #1 {\large \ding{55}};}
\renewcommand{\circAT}[1]{\draw[thick] #1 circle (0.13);}

\renewcommand*\bblock[2]{\draw #1 [thin] rectangle ++(0.5,0.5) node [pos=.5] {\tiny{#2} }; }

\renewcommand{\VRule}[1][\arrayrulewidth]{\vrule width #1}

\renewcommand{\trenner}{\ensuremath{\tab[0.05cm]\Big|\tab[0.05cm]}}
