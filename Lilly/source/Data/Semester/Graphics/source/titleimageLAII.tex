\newcommand\placeheart[1]{
  \begin{tikzpicture}
    \begin{axis}[
        view={0}{10},
        axis equal,
        axis lines=none,
        colormap name =pink,
        scale=1.6
      ]
      \addplot3[
        surf,
        shader=faceted,
        samples=#1,
        domain=0:2*pi,y domain=0:2*pi,
        z buffer=sort,
        opacity=0.35
      ]
      (
        {(sin(deg(x)))^3*cos(deg(y))},
        {(sin(deg(x)))^3*sin(deg(y))},
        {(13*cos(deg(x))-5*cos(2*deg(x))-2*cos(3*deg(x))-cos(4*deg(x)))/16}
      );
    \end{axis}
  \end{tikzpicture}
}

        opacity=0.8
    ]
    {sin(deg(sqrt(x^2+y^2)))/sqrt(x^2+y^2)};
    \end{axis}
    \end{tikzpicture}
