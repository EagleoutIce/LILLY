% #1 at; #2 color; #3: width (\p+3\p+\p), #4: depth (20\p), #5 count caps x, #6 count caps y, #7 foreground cheats :D
\def\BrickGeneric#1#2#3#4#5#6#7{
    % front -| ++(-1\p,1\p,0) -| ++(-3\p,-1\p,0)
    \draw[fill=#2] (#1) -| ++(#3,6\p,0) -| cycle;
    % 3d back
    \draw[fill=#2] (#1)++(#3,0,0) -- ++(0,0,-#4) |- ++(-#3,6\p,0) -- ++(0,0,#4) -- ++(#3,0,0) -- ++(0,-6\p,0);
    % strut line
    \draw (#1)++(#3,6\p,0) -- ++(0,0,-#4);
    % caps
    \foreach \x in {#5,...,1} {
        \foreach \y in {#6,...,1} {
            \node[cylinder,draw,fill=#2,shape border rotate=130,text width=150\p,aspect =.3, scale=0.5] at ($(#1)+(2.5\p+\x*5\p-5\p,6\p,-3\p-\y*5\p+5\p)$) {};%-\i/25
        }
    }
    \ifdobuffer
        \edef\tmpXContainer{\noexpand\pushList{colorBuffer\BufferC}{#1/#2/{#7}}}\tmpXContainer%
    \else
        \begin{pgfonlayer}{foreground}
            \foreach \x/\y in {#7} {
                \node[cylinder,draw,fill=#2,shape border rotate=130,text width=150\p,aspect =.3, scale=0.5] at ($(#1)+(2.5\p+\x*5\p-5\p,6\p,-3\p-\y*5\p+5\p)$) {};%-\i/25
            }
        \end{pgfonlayer}
    \fi
}


\def\PrintBuffer#1{%
    \csname getcolorBuffer#1\endcsname\relax
    \foreach \pos/\col/\lst in \lillyxlist {
        \foreach \x/\y in \lst {
            % \typeout{X: \x, Y: \y, pos: \pos, col: \col}
            \node[cylinder,draw,fill=\col,shape border rotate=130,text width=150\p,aspect =.3, scale=0.5] at ($(\pos)+(2.5\p+\x*5\p-5\p,6\p,-3\p-\y*5\p+5\p)$) {};%-\i/25
        }
    }
    \setList{colorBuffer#1}{}% clear
    \gdef\lillyxlist{}
}
