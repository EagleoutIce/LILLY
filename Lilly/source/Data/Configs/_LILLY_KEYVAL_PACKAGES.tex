% Diese Optionen erlauben es die Pakete, die Lilly einbindet, zu steuern.

% ifLILLY@math: Control the math-module
\DeclareBoolOption[true]{math}\DeclareComplementaryOption{nomath}{math}

% ifLILLY@listings: Control the listings-module
\DeclareBoolOption[true]{listings}\DeclareComplementaryOption{nolistings}{listings}

% ifLILLY@runtimes: Control the runtimes-module
\DeclareBoolOption[true]{runtimes}\DeclareComplementaryOption{noruntimes}{runtimes}

% ifLILLY@graphics: Control the graphics-module
\DeclareBoolOption[true]{graphics}\DeclareComplementaryOption{nographics}{graphics}

% ifLILLY@presenter: Control the presenter-module
\DeclareBoolOption[true]{presenter}\DeclareComplementaryOption{nopresenter}{presenter}
