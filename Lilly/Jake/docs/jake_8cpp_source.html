319 status_t ld_settings(int n /* = argc */, const char** argv) {
320  for(; n > 0; n--) {
321  if(!in_str(argv[n],ARG_PATTERN)){ // not a single argument
322  if(!in_str(argv[n],TEX_PATTERN)) { // not a .tex file
323  settings["operation"] = TO_DATA(argv[n]);
324  } else { // is a .tex file
325  settings["operation"] = "file_compile";
326  settings["file"] = TO_DATA(argv[n]);
327  }
328  } else { // is a single argument
329  if(!in_str(argv[n], ASS_PATTERN)) { // not an assignment
330  if(settings.find(argv[n] + 1 /* offset for '-' */) != settings.end()
331  && settings[argv[n]+1] == "false") { // valid bool-setting
332  settings[argv[n]+1] = "true";
333  } else { // no valid bool-setting
334  er_unknown_setting(argv[n]+1);
335  }
336  } else { // is an assignment
337  std::string s{argv[n]+1};
338  settings[s.substr(0,s.find("="))] = s.substr(s.find("=")+1);
339  }
340  }
341  }
342 }
