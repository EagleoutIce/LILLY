250 status_t ld_settings(int n /* = argc */, const char** argv) {
251  for(; n > 0; n--) {
252  if(!in_str(argv[n],ARG_PATTERN)){ // not a single argument
253  if(!in_str(argv[n],TEX_PATTERN)) { // not a .tex file
254  settings["operation"] = TO_DATA(argv[n]);
255  } else { // is a .tex file
256  settings["operation"] = "file_compile";
257  settings["file"] = TO_DATA(argv[n]);
258  }
259  } else { // is a single argument
260  if(!in_str(argv[n], ASS_PATTERN)) { // not an assignment
261  if(settings.find(argv[n] + 1 /* offset for '-' */) != settings.end()
262  && settings[argv[n]+1] == "false") { // valid bool-setting
263  settings[argv[n]+1] = "true";
264  } else { // no valid bool-setting
265  er_unknown_setting(argv[n]+1);
266  }
267  } else { // is an assignment
268  std::string s{argv[n]+1};
269  settings[s.substr(0,s.find("="))] = s.substr(s.find("=")+1);
270  }
271  }
272  }
273 }
