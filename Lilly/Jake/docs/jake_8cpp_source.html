365 status_t ld_settings(int n /* = argc */, const char** argv) {
366  for(; n > 0; n--) {
367  if(!in_str(argv[n],ARG_PATTERN)){ // not a single argument
368  if(!in_str(argv[n],TEX_PATTERN)) { // not a .tex file
369  settings["operation"] = TO_DATA(argv[n]);
370  } else { // is a .tex file
371  settings["operation"] = "file_compile";
372  settings["file"] = TO_DATA(argv[n]);
373  }
374  } else { // is a single argument
375  if(!in_str(argv[n], ASS_PATTERN)) { // not an assignment
376  if(settings.find(argv[n] + 1 /* offset for '-' */) != settings.end()
377  && settings[argv[n]+1] == "false") { // valid bool-setting
378  settings[argv[n]+1] = "true";
379  } else { // non valid bool-setting
380  er_unknown_setting(argv[n]+1);
381  }
382  } else { // is an assignment
383  std::string s{argv[n]+1};
384  if(in_str(argv[n], ASA_PATTERN)) { // is an addition statement
385  settings[s.substr(0,s.find(ASA_PATTERN))] += s.substr(s.find(ASA_PATTERN)+std::string(ASA_PATTERN).length());
386  } else { // normal statement
387  settings[s.substr(0,s.find(ASS_PATTERN))] = s.substr(s.find(ASS_PATTERN)+std::string(ASS_PATTERN).length());
388  }
389  }
390  }
391  }
392  return EXIT_SUCCESS;
393 }
